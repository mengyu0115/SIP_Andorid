```java
public void prewarm(int deviceId) throws FrameGrabber.Exception {
    if (isPrewarmed && grabber != null) {
        log.info("摄像头已预热，跳过");
        return;
    }

    log.info("🔥 预热摄像头: 设备 {}", deviceId);
    long startTime = System.currentTimeMillis();

    // 步骤1：创建grabber对象
    grabber = new OpenCVFrameGrabber(deviceId);
    long step1Time = System.currentTimeMillis();
    log.info("⏱️ [预热] 步骤1完成（创建grabber）: {} ms", step1Time - startTime);

    // 步骤2：设置参数
    grabber.setImageWidth(VIDEO_WIDTH);
    grabber.setImageHeight(VIDEO_HEIGHT);
    grabber.setFrameRate(FRAME_RATE);
    long step2Time = System.currentTimeMillis();
    log.info("⏱️ [预热] 步骤2完成（设置参数）: {} ms", step2Time - step1Time);

    // 步骤3：启动grabber（耗时最长）
    grabber.start();
    long step3Time = System.currentTimeMillis();
    log.info("⏱️ [预热] 步骤3完成（grabber.start）: {} ms", step3Time - step2Time);

    try {
        // 步骤4：抓取测试帧
        Frame testFrame = grabber.grab();
        long step4Time = System.currentTimeMillis();
        log.info("⏱️ [预热] 步骤4完成（grab测试帧）: {} ms", step4Time - step3Time);

        if (testFrame != null) {
            log.info("✅ 摄像头预热成功: 测试帧 {}x{}", testFrame.imageWidth, testFrame.imageHeight);
        }
    } catch (Exception e) {
        log.warn("⚠️ 预热时抓取测试帧失败（可忽略）: {}", e.getMessage());
    }

    isPrewarmed = true;
    long totalTime = System.currentTimeMillis() - startTime;
    log.info("✅ 摄像头预热完成: {}x{} @ {}fps (grabber已启动，可直接使用), 总耗时: {} ms",
        VIDEO_WIDTH, VIDEO_HEIGHT, FRAME_RATE, totalTime);
}
```
